class Animals:
    def __init__(self, species, name, color):
        self.species = species
        self.name = name
        self.color = color
    def __str__(self):
        return f"{self.species} named {self.name} is {self.color} in color."
import time

# Get the current local time as a struct_time object
current_time = time.localtime()

# Print the entire object
print(f"Poora struct_time object: {current_time}")

# Accessing individual parts (attributes)
print("--- Individual Parts ---")
print(f"Saal (Year): {current_time.tm_year}")
print(f"Mahina (Month): {current_time.tm_mon}")
print(f"Din (Day): {current_time.tm_mday}")
print(f"Ghanta (Hour): {current_time.tm_hour}")
print(f"Minute: {current_time.tm_min}")

# Day of the week (0 is Monday, 1 is Tuesday, etc.)
print(f"Hafte ka din (0=Mon): {current_time.tm_wday}")
